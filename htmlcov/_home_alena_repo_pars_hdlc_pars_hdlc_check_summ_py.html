def checksum(octets): 

"""Compute Fast Frame Check Sequence (FCS) and Header Check Sequence (HCS) 

Note: borrowed from CMP 

 

Args: 

octets 

octet string in hex 

 

Returns: 

calculated checksum 

 

Based on rfc1662 

""" 

value = 0xFFFF 

fcs_table = CS_TABLE 

for octet in octets: 

value = (value >> 8) ^ fcs_table[(value ^ ord(octet)) & 0xFF] 

 

return value ^ 0xFFFF
